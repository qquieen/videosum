### 4.2 视频总结页 (`Summary.vue`)

```vue
<template>
  <div class="summary-page">
    <el-row :gutter="20">
      <!-- 左侧：输入区 -->
      <el-col :span="8">
        <el-card class="input-card">
          <template #header>
            <span>输入</span>
          </template>
          
          <el-form :model="form">
            <el-form-item label="视频URL或文件路径">
              <el-input 
                v-model="form.url"
                type="textarea"
                :rows="3"
                placeholder="输入B站、YouTube链接，或本地文件路径..."
              />
            </el-form-item>
            
            <el-form-item label="上传文件">
              <el-upload
                drag
                :auto-upload="false"
                :on-change="handleFileChange"
              >
                <el-icon class="el-icon--upload"><upload-filled /></el-icon>
                <div class="el-upload__text">
                  拖拽文件到此处，或<em>点击上传</em>
                </div>
              </el-upload>
            </el-form-item>
            
            <el-form-item label="输出语言">
              <el-radio-group v-model="form.language">
                <el-radio label="zh">中文</el-radio>
                <el-radio label="en">English</el-radio>
                <el-radio label="de">Deutsch</el-radio>
              </el-radio-group>
            </el-form-item>
            
            <el-button 
              type="primary" 
              size="large"
              :loading="loading"
              @click="handleStart"
            >
              开始总结
            </el-button>
          </el-form>
          
          <el-progress 
            v-if="loading"
            :percentage="progress" 
            :status="progressStatus"
          />
          
          <el-alert 
            v-if="status"
            :title="status"
            :type="statusType"
            show-icon
          />
        </el-card>
      </el-col>
      
      <!-- 右侧：输出区 -->
      <el-col :span="16">
        <el-tabs v-model="activeTab">
          <el-tab-pane label="总结结果" name="summary">
            <div class="summary-content" v-html="summaryHtml" />
            <el-button-group>
              <el-button @click="handleCopy">复制</el-button>
              <el-button @click="handleExport">导出</el-button>
            </el-button-group>
          </el-tab-pane>
          
          <el-tab-pane label="问答助手" name="qa">
            <div class="chat-container">
              <div class="chat-messages">
                <div 
                  v-for="(msg, index) in chatHistory" 
                  :key="index"
                  :class="['message', msg.role]"
                >
                  <div class="message-content">{{ msg.content }}</div>
                </div>
              </div>
              
              <div class="chat-input">
                <el-input 
                  v-model="question"
                  placeholder="基于视频内容提问..."
                  @keyup.enter="handleQA"
                >
                  <template #append>
                    <el-button @click="handleQA">发送</el-button>
                  </template>
                </el-input>
              </div>
            </div>
          </el-tab-pane>
          
          <el-tab-pane label="课件预览" name="courseware">
            <el-empty description="课件生成功能开发中..." />
          </el-tab-pane>
        </el-tabs>
      </el-col>
    </el-row>
  </div>
</template>

<script setup>
import { ref, reactive } from 'vue'
import { summarize, qa } from '@/api'

const form = reactive({
  url: '',
  language: 'zh'
})

const loading = ref(false)
const progress = ref(0)
const status = ref('')
const statusType = ref('info')
const summaryHtml = ref('')
const activeTab = ref('summary')
const question = ref('')
const chatHistory = ref([])

const handleStart = async () => {
  loading.value = true
  progress.value = 0
  status.value = '处理中...'
  statusType.value = 'info'
  
  try {
    const result = await summarize(form.url, form.language)
    summaryHtml.value = result.summary
    status.value = `完成 - 耗时 ${result.processing_time.toFixed(1)}s`
    statusType.value = 'success'
  } catch (error) {
    status.value = `失败: ${error.message}`
    statusType.value = 'error'
  } finally {
    loading.value = false
  }
}

const handleQA = async () => {
  if (!question.value.trim()) return
  
  chatHistory.value.push({ role: 'user', content: question.value })
  
  try {
    const result = await qa(currentTaskId, question.value)
    chatHistory.value.push({ role: 'assistant', content: result.answer })
  } catch (error) {
    chatHistory.value.push({ role: 'assistant', content: `错误: ${error.message}` })
  }
  
  question.value = ''
}
</script>
```

## 6. 状态管理

```javascript
// stores/app.js
import { defineStore } from 'pinia'

export const useAppStore = defineStore('app', {
  state: () => ({
    theme: 'light',
    language: 'zh',
    currentTaskId: null,
    tasks: []
  }),
  
  actions: {
    setTheme(theme) {
      this.theme = theme
    },
    
    setCurrentTask(taskId) {
      this.currentTaskId = taskId
    },
    
    addTask(task) {
      this.tasks.push(task)
    }
  }
})
```

## 7. API调用封装

```javascript
// api/index.js
import axios from 'axios'

const api = axios.create({
  baseURL: 'http://localhost:8000/api',
  timeout: 30000
})

export const summarize = async (url, language) => {
  const response = await api.post('/video/summarize', { url, language })
  return response.data
}

export const qa = async (taskId, question) => {
  const response = await api.post('/video/qa', { task_id: taskId, question })
  return response.data
}

export const getConfig = async () => {
  const response = await api.get('/config')
  return response.data
}

export const updateConfig = async (config) => {
  const response = await api.post('/config', config)
  return response.data
}

export const getTasks = async () => {
  const response = await api.get('/tasks')
  return response.data
}

export const deleteTask = async (taskId) => {
  const response = await api.delete(`/tasks/${taskId}`)
  return response.data
}
```
